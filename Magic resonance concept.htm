Here's a userJS script that you can use to save the conversation in markdown format. It extracts the conversation from the DOM, formats it, and allows you to download it as a .md file.

UserJS Script:

javascript
// ==UserScript==
// @name         Save Chat Conversation as Markdown
// @namespace    http://tampermonkey.net/
// @version      0.1
// @description  Save chat conversations from the page as a markdown file
// @author       Faizal
// @match        *://*/*
// @grant        GM_download
// ==/UserScript==

(function() {
    'use strict';

    // Function to format the conversation into markdown
    function formatConversationToMarkdown() {
        let markdownContent = "# Chat Conversation\n\n";
        let messages = document.querySelectorAll('.message-class'); // Adjust selector based on your page structure

        messages.forEach((msg, index) => {
            let username = msg.querySelector('.username-class') ? msg.querySelector('.username-class').textContent : 'Unknown';
            let timestamp = msg.querySelector('.timestamp-class') ? msg.querySelector('.timestamp-class').textContent : 'No Timestamp';
            let messageText = msg.querySelector('.message-text-class') ? msg.querySelector('.message-text-class').textContent : '';

            markdownContent += `**${username}** - ${timestamp}\n\n${messageText}\n\n---\n`;
        });

        return markdownContent;
    }

    // Function to save markdown as .md file
    function saveMarkdownFile(content) {
        const blob = new Blob([content], { type: 'text/markdown' });
        const fileName = `Chat_Conversation_${new Date().toISOString()}.md`;

        GM_download({
            url: URL.createObjectURL(blob),
            name: fileName,
            saveAs: true
        });
    }

    // Add button to trigger saving the conversation
    const button = document.createElement('button');
    button.textContent = 'Save Chat as Markdown';
    button.style.position = 'fixed';
    button.style.top = '10px';
    button.style.right = '10px';
    button.style.padding = '10px';
    button.style.backgroundColor = '#4CAF50';
    button.style.color = 'white';
    button.style.border = 'none';
    button.style.borderRadius = '5px';
    button.style.cursor = 'pointer';
    button.addEventListener('click', function() {
        const markdownContent = formatConversationToMarkdown();
        saveMarkdownFile(markdownContent);
    });

    // Append the button to the body
    document.body.appendChild(button);
})();

How to Use:

  1. Install Tampermonkey: You’ll need Tampermonkey or any userscript manager to run this script. Install it from the browser extension store if you don’t have it.

  2. Add the Script: Create a new script in Tampermonkey and paste the code above into the editor.

  3. Modify Selectors: Update the .message-class, .username-class, .timestamp-class, and .message-text-class selectors to match your specific website's DOM structure. These classes should target the message content, user names, timestamps, and text respectively.

  4. Save the Conversation: Once the script is running on the page, a button will appear in the top-right corner. Click on it to download the conversation in markdown format.
